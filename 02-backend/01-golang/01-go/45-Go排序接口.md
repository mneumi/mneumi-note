## 官方接口

```go
type Interface interface {
    Len() int
    Less(i, j int) bool
    Swap(i, j int)
}
```



## 实现案例

```go
import "sort"

type Course struct {
    Name string
    Price int
    Url string
}

type Courses []Course

func (c Courses) Len() int {
    return len(c)
}

func (c Courses) Less(i, j int) bool {
    return c[i].Price < c[j].Price
}

func (c Courses) Swap(i, j int) {
    c[i], c[j] = c[j], c[i]
}

func main() {
    courses := Courses{
        Course{"django", 300, ""},
        Course{"scrapy", 100, ""},
        Course{"tornado", 200, ""}
    }
    sort.Sort(courses)
}
```
