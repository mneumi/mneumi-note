1. 下列输出结果是什么

   ```go
   package main
   
   import "fmt"
   
   type student struct {
     Name string
     Age int
   }
   
   func main() {
     m := make(map[string]*student)
     
     stus := []student {
       {Name: "zhou",Age: 22},
       {Name: "li", Age: 23},
       {Name: "wang", Age: 24},
     }
     
     for _, stu := range stus {
       m[stu.Name] = &stu
     }
     
     for _, v := range m {
       fmt.Println(v.Name, " ", v.Age)
     }
   }
   ```
